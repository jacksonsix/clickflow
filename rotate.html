//////////css [rotate-on-click] { height: 50px; width: 50px; background: black; } //////////////js function rotate(deg, elem) { $(elem).css({ webkitTransform: 'rotate('+deg+'deg)', mozTransform: 'rotate('+deg+'deg)', msTransform: 'rotate('+deg+'deg)', oTransform: 'rotate('+deg+'deg)', transform: 'rotate('+deg+'deg)' }); } function addRotateOnClick($elems) { $elems.each(function(i, elem) { var deg = 0; $(elem).click(function() { deg+= parseInt($(this).attr('rotate-on-click'), 10); console.log(deg); rotate(deg, this); }); }); } addRotateOnClick($('[rotate-on-click]'));
